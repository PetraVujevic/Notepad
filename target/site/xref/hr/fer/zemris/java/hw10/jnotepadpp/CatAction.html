18  /**
19   * Holds action that is in charge for copying and deleting selected text area.
20   * 
21   * @author petra
22   *
23   */
24  public class CatAction extends LocalizableAction {
25  	// default serial version id
26  	private static final long serialVersionUID = 1L;
27  	// editors that notepad contains
28  	private List<Editor> editors = new ArrayList<>();
29  	// cardinal number of current tab
30  	private int currentTab;
31  	// notepad
32  	JNotepadPP notepad;
33  
34  	/**
35  	 * Sets action's name and mnemonic key, initializes variables.
36  	 * 
37  	 * @param key
38  	 *            String to be translated
39  	 * @param lp
40  	 *            Localization provider
41  	 * @param editors
42  	 *            Editors opened in current JNotepadPP
43  	 * @param currentTab
44  	 *            Currently opened tab
45  	 * @param notepad
46  	 *            JNotepadPP
47  	 */
48  	public CatAction(String key, ILocalizationProvider lp,
49  			List<Editor> editors, int currentTab, JNotepadPP notepad) {
50  		super(key, lp);
51  		this.editors = editors;
52  		this.currentTab = currentTab;
53  		this.notepad = notepad;
54  		putValue(Action.ACCELERATOR_KEY, KeyStroke.getKeyStroke("control X"));
55  		putValue(Action.MNEMONIC_KEY, KeyEvent.VK_X);
56  	}
57  
58  	@Override
59  	public void actionPerformed(ActionEvent e) {
60  		JTextArea editor = editors.get(currentTab).getEditor();
61  		Document doc = editor.getDocument();
62  		int pocetak = Math.min(editor.getCaret().getDot(), editor.getCaret()
63  				.getMark());
64  		int duljina = Math.max(editor.getCaret().getDot(), editor.getCaret()
65  				.getMark())
66  				- pocetak;
67  		try {
68  			notepad.paste = doc.getText(pocetak, duljina);
69  			doc.remove(pocetak, duljina);
70  		} catch (BadLocationException e1) {
71  			JOptionPane.showMessageDialog(editor,
72  					"It is not possible to get your content of the document.");
73  		}
74  	}
75  }
