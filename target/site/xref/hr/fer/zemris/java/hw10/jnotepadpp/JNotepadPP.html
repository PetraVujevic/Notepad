143 			}

146 
147 			@Override
148 			public void caretUpdate(CaretEvent e) {
149 				length = editor.getDocument().getLength();
150 				int caretpos = editor.getCaretPosition();
151 				try {
152 					ln = editor.getLineOfOffset(caretpos);
153 					col = caretpos - editor.getLineStartOffset(ln);
154 					int pocetak = Math.min(editor.getCaret().getDot(), editor
155 							.getCaret().getMark());
156 					selected = Math.max(editor.getCaret().getDot(), editor
157 							.getCaret().getMark())
158 							- pocetak;
159 				} catch (BadLocationException e1) {
160 					JOptionPane.showMessageDialog(editor,
161 							"It is not possible to get current line and row.");
162 				}
163 				ln += 1;
164 				numbers.repaint();
165 			}
166 		});
167 		editors.add(new Editor(editor));
168 		tabs.addTab("New file", new ImageIcon("img/green.png"),
169 				new JScrollPane(editor), "New file");
170 		getContentPane().add(tabs, BorderLayout.CENTER);
171 		numbers = new JPanel(new GridLayout(1, 2));
172 		numbers.setBorder(BorderFactory.createMatteBorder(3, 1, 1, 1,
173 				Color.gray));
174 
175 		timeLabel = new Time();
176 		timeLabel.setText("Currently :");
177 		timeLabel.setForeground(Color.black);
178 		timeLabel.setOpaque(true);
179 		lenLabel = new Labela();
180 		lenLabel.setText("Currently :");
181 		lenLabel.setForeground(Color.black);

300 	/**
301 	 * Sorts selected lines. Type of sorting is conditioned by sorter given as
302 	 * parameter.
303 	 * 
304 	 * @param sorter
305 	 *            Determines the way of sorting that can be
306 	 *            ascending/descending/unique. Unique is actually not sorting,
307 	 *            it removes duplicated lines.
308 	 */
309 	public static void sort(ISort sorter) {
310 		Locale hrLocale = new Locale(jezik);
311 		Collator collator = Collator.getInstance(hrLocale);
312 		JTextArea editor = editors.get(currentTab).getEditor();
313 		try {
314 			int selectedBegin = Math.min(editor.getCaret().getDot(), editor
315 					.getCaret().getMark());
316 			int lineBegin = editor.getLineOfOffset(selectedBegin);
317 			int offsetBegin = editor.getLineStartOffset(lineBegin);
318 
319 			int selectedEnd = Math.max(editor.getCaret().getDot(), editor
320 					.getCaret().getMark());
321 			int lineEnd = editor.getLineOfOffset(selectedEnd);
322 			int offsetEnd = editor.getLineEndOffset(lineEnd);
323 			String toSort = editor.getDocument().getText(offsetBegin,
324 					offsetEnd - offsetBegin);
325 			List<String> lines = Arrays.asList(toSort.split("\\n"));
326 
327 			lines = sorter.sortLines(lines, collator);
328 
329 			String toInsert = String.join("\n", lines) + "\n";
330 			editor.getDocument().remove(offsetBegin, offsetEnd - offsetBegin);
331 			editor.getDocument().insertString(offsetBegin, toInsert, null);
332 		} catch (BadLocationException e1) {
333 			JOptionPane.showMessageDialog(editor,
334 					"It is not possible to get your content of the document.");
335 		}
336 	}

339 	 * Extracts selected area.
340 	 * 
341 	 * @param changer
342 	 *            Determines the way selected area is going to be changed to :
343 	 *            upper case/ lower case/ inverted.
344 	 */
345 	public static void textoToChangeCase(IChangeCase changer) {
346 		JTextArea editor = editors.get(currentTab).getEditor();
347 		Document doc = editor.getDocument();
348 		int pocetak = Math.min(editor.getCaret().getDot(), editor.getCaret()
349 				.getMark());
350 		int duljina = Math.max(editor.getCaret().getDot(), editor.getCaret()
351 				.getMark())
352 				- pocetak;
353 		try {
354 			String text = doc.getText(pocetak, duljina);
355 			text = changeCase(text, changer);
356 			doc.remove(pocetak, duljina);
357 			doc.insertString(pocetak, text, null);
358 		} catch (BadLocationException ignorable) {
359 		}
360 	}

362 	/**
363 	 * Changes string in a way that is determined by parameter changer.
364 	 * 
365 	 * @param text
366 	 *            Text to be changed
367 	 * @param changer
368 	 *            Upper case/Lower case/Toggle case
369 	 * @return Changed string.
370 	 */
371 	private static String changeCase(String text, IChangeCase changer) {
372 		char[] znakovi = text.toCharArray();
373 		for (int i = 0; i < znakovi.length; i++) {
374 			char c = znakovi[i];
375 			znakovi[i] = changer.changeCase(c);
376 		}
377 		return new String(znakovi);
378 	}

380 	/**
381 	 * Asks for permission is current file is about to overwrite existing file
382 	 * and acts depending on user's answer (overwrites or discards current
383 	 * file).
384 	 */
385 	protected void save() {
386 		Path openedFilePath = editors.get(currentTab).getPath();
387 
388 		if (openedFilePath == null) {
389 			JFileChooser fc = new JFileChooser();
390 			fc.setDialogTitle("Save document");
391 			if (fc.showSaveDialog(JNotepadPP.this) != JFileChooser.APPROVE_OPTION) {
392 				JOptionPane.showMessageDialog(JNotepadPP.this,
393 						"Nista nije pohranjeno", "Poruka",
394 						JOptionPane.INFORMATION_MESSAGE);
395 				return;
396 			}
397 			Path file = fc.getSelectedFile().toPath();
398 			if (Files.exists(file)) {
399 				int rez = JOptionPane
400 						.showConfirmDialog(
401 								JNotepadPP.this,
402 								"Odabrana datoteka ("
403 										+ file
404 										+ ") vec postoji. Jeste li SIGURNI da ju zelite pregaziti ?",
405 								"Upozorenje", JOptionPane.YES_NO_OPTION,
406 								JOptionPane.WARNING_MESSAGE);
407 				if (rez != JOptionPane.YES_OPTION) {
408 					return;
409 				}
410 			}
411 			openedFilePath = file;
412 		}
413 		try {
414 			JTextArea editor = editors.get(currentTab).getEditor();
415 			Files.write(openedFilePath,
416 					editor.getText().getBytes(StandardCharsets.UTF_8));
417 			editors.get(currentTab).setOriginal(editor);
418 			String path = openedFilePath.toString();
419 			editors.get(currentTab).setPath(new File(path).toPath());
420 			int n = path.lastIndexOf(File.separatorChar);
421 			String title = path.substring(n + 1);
422 			String file = path.substring(0, n);
423 			tabs.setTitleAt(currentTab, file + " - " + title);
424 			tabs.setToolTipTextAt(currentTab, path);
425 			tabs.setIconAt(currentTab, new ImageIcon("img/green.png"));
426 		} catch (IOException e1) {
427 			JOptionPane.showMessageDialog(JNotepadPP.this,
428 					"Pogreska pri zapisivanju datoteke " + openedFilePath
429 							+ ": " + e1.getMessage() + ".", "Pogreska",
430 					JOptionPane.ERROR_MESSAGE);
431 			return;
432 		}
433 	}

435 	/**
436 	 * Asks if user wants to save document before closing, saves document if
437 	 * requested and closes document.
438 	 */
439 	protected void close() {
440 		JTextArea editor = editors.get(currentTab).getEditor();
441 		JTextArea original = editors.get(currentTab).getOriginal();
442 		if (!editor.getText().equals(original.getText())) {
443 			int rez = JOptionPane.showConfirmDialog(JNotepadPP.this,
444 					"Zelite li spremiti prije zatvaranja ?", "Upozorenje",
445 					JOptionPane.YES_NO_OPTION, JOptionPane.WARNING_MESSAGE);
446 			if (rez == JOptionPane.YES_OPTION) {
447 				save();
448 			}
449 		}
450 		editors.remove(currentTab);
451 		tabs.remove(currentTab);
452 	}

488 	/**
489 	 * Creates one independent thread for time display and displays time in the
490 	 * right bottom part of the main window.
491 	 * 
492 	 * @author petra
493 	 *
494 	 */
495 	private class Time extends JLabel {
496 		public Time() {
497 			Thread t = new Thread(() -> {
498 				while (true && !stopTime) {
499 					SwingUtilities.invokeLater(() -> {
500 						repaint();
501 					});
502 					try {
503 						Thread.sleep(500);
504 					} catch (Exception ignorable) {
505 
506 					}
507 				}
508 			});
509 
510 			t.setDaemon(true);
511 			t.start();
512 		}
513 
514 		@Override
515 		protected void paintComponent(Graphics g) {
516 			super.paintComponents(g);
517 			FontMetrics fm = g.getFontMetrics();
518 			int yDescriptionLen = fm.stringWidth(length + "");
519 			Dimension size = getSize();
520 			setForeground(Color.black);
521 			g.drawString(new Date().toString(), (int) (size.getHeight() / 2)
522 					- yDescriptionLen / 2, 12);
523 		}
524 	}
