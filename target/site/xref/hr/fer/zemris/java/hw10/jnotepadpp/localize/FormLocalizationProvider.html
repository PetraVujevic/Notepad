3   import java.awt.event.WindowAdapter;
4   import java.awt.event.WindowEvent;
5   
6   import javax.swing.JFrame;
7   
8   /**
9    * Class derived from LocalizationProviderBridge; in its constructor it
10   * registers itself as a WindowListener to its JFrame; when frame is opened, it
11   * calls connect and when frame is closed, it calls disconnect.
12   * 
13   * @author petra
14   *
15   */
16  public class FormLocalizationProvider extends LocalizationProviderBridge {
17  
18  	/**
19  	 * When frame is opened, it calls connect and when frame is closed, it calls
20  	 * disconnect.
21  	 * 
22  	 * @param lp
23  	 *            Localization provider
24  	 * @param jf
25  	 *            Frame
26  	 */
27  	public FormLocalizationProvider(ILocalizationProvider lp, JFrame jf) {
28  		super(lp);
29  		jf.addWindowListener(new WindowAdapter() {
30  			@Override
31  			public void windowOpened(WindowEvent e) {
32  				connect();
33  			}
34  
35  			@Override
36  			public void windowClosed(WindowEvent e) {
37  				disconnect();
38  			}
39  		});
40  	}
41  }
